version: '3.8'
services:
  mongo:
    image: mongo:6
    container_name: mongo
    restart: always
    environment:
      MONGO_INITDB_ROOT_USERNAME: root
      MONGO_INITDB_ROOT_PASSWORD: example
    ports:
      - "27017:27017"
    volumes:
      - mongo_data:/data/db

  nest-api:
    build:
      context: ./apps/nest-api
    container_name: nest-api
    restart: always
    environment:
      MONGODB_URI: mongodb://root:example@mongo:27017/nest_next_copilot?authSource=admin
      PORT: 3333
    ports:
      - "3333:3333"
    depends_on:
      - mongo

  next-app:
    build:
      context: ./apps/next-app
    container_name: next-app
    restart: always
    environment:
      NEXT_PUBLIC_API_URL: http://nest-api:3333
      # (add other envs as needed)
    ports:
      - "3000:3000"
    depends_on:
      - nest-api
